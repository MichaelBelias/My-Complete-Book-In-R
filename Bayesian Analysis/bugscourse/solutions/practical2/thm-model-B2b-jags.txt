model {
tau <- 1/pow(5, 2)

for (i in 1:2) {
  y[i] ~ dnorm(theta, tau)
}

# vague normal prior
theta ~ dnorm(0, 0.000001)

# predictions and probabilities
ypred ~ dnorm(theta, tau)
py.130 <- step(ypred - 130)    # indicator of whether ypred > 130
pmean.130 <- step(theta - 130)    # indicator of whether zone mean > 130
}
