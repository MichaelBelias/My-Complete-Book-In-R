model {
for (i in 1:N){
  r[i] ~ dbin(p[i], n[i])

  # Original, centred version
  # logit(p[i]) <- alpha + beta * (x[i] - mean(x[]))

  # Probit, centred version
  probit(p[i]) <- alpha + beta * (x[i] - mean(x[]))

  rhat[i] <- n[i] * p[i]
  phat[i] <- r[i] / n[i]
}
beta ~ dnorm(0, 0.00001)
alpha ~ dnorm(0, 0.00001)
}
